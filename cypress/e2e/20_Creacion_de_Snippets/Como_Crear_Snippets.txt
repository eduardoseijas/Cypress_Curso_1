En el buscador, del visual, donde esta la lupa, colocamos:
>snippets

Seleccionamos configuration snippets y luego javascript, nos abrira un archivo con esta info:

{
	// Place your snippets for javascript here. Each snippet is defined under a snippet name and has a prefix, body and 
	// description. The prefix is what is used to trigger the snippet and the body will be expanded and inserted. Possible variables are:
	// $1, $2 for tab stops, $0 for the final cursor position, and ${1:label}, ${2:another} for placeholders. Placeholders with the 
	// same ids are connected.
	// Example:
	// "Print to console": {
	// 	"prefix": "log",
	// 	"body": [
	// 		"console.log('$1');",
	// 		"$2"
	// 	],
	// 	"description": "Log output to console"
	// }

	"texto_visible":{
		"prefix": "c_texto_visible",
		"body":[
		"cy.get('$1').should('be.visible').type('$2')"
		//texto visible
		],
		"description": "Texto Visible"
	},

	"Plantilla_Cypress":{
		"prefix": "c_plantilla_cypress",
		"body":[
			"///<reference types = 'Cypress'/>",
//Esto de la linea 1 es para que funcionen los comandos

            "require('cypress-plugin-tab');", // este se instala npm install -D cypress-plugin-tab
            "require('cypress-xpath')", // este se instala npm install -D cypress-xpath
            "import 'cypress-file-upload';",
            "require('@4tw/cypress-drag-drop')",

			"describe('${1:descripcion}', ()=>{",
				"\t\tit('${2:test}', function(){",
					"\t\tcy.visit('${3:url}')",
					"\t\t\tcy.title().should('eq','${4:titulo}')",
			
				"})",
			"})",
		],
		"description": "Pantilla Cypress"
	},

}
